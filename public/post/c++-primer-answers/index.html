C++ Primer Exercises Answers

Section 2.3.1

Exercise 2.15

int ival = 1.01; // correct
int &rval1 = 1.01; // wrong, type error
int &rval2 = ival; // correct
int &rval3; // not initialized

Exercise 2.16

int i = 0, &r1 = i; 
double d = 0, &r2 = d;
    
r2 = 3.14159; // correct
r2 = r1; // correct
i = r2; // correct
r1 = d; // correct

Exercise 2.17

int i, &ri = i;
i = 5;
ri = 10;
std::cout << i << " " << ri << std::endl;

Section 2.3.2

Exercise 2.18

int a = 1;
int b = 2;
int *p = &a;
p = &b;
*p = 3;
std::cout << *p;

Exercise 2.20

int i = 42;
int *p1 = &i;
*p1 = *p1 * *p1;
std::cout << i;

Exercise 2.21

int i = 0;
double *dp = &i; // error
int *ip = i; // error
int *p = &i; // correct

Section 2.4.2

Exercise 2.27

int a = -1, &b = 0; // wrong, reference to a constant 0
int c = 9, *const d = &c; // correct
const int e = -1, &f = 0; // correct
const int *const g = &a; // correct
const int *h = &a; // correct
const int &const i; // wrong
const int z = a, &y = a; // correct

Section 2.4.3

Exercise 2.30

int i = 9;
const int v2 = 0;
int v1 = v2;
int *p1 = &v1, &r1 = v1;
const int *p2 = &v2, *const p3 = &i, &r2 = v2;

r1 = v2; // correct
p1 = p2; // wrong
p2 = p1; // correct
p1 = p3; // wrong
p2 = p3; // correct
